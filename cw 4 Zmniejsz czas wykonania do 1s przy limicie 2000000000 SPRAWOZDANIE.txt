            int dana = 2000000;
            var miernik = new System.Diagnostics.Stopwatch();
            miernik.Start();
            long suma = Suma(dana);
            miernik.Stop();
            string wiadomosc = String.Format("Suma {1} liczb wynosi: {0}, Zajê³o: {2}", suma, dana, miernik.Elapsed);

        public static long Suma(long limit)
        {
            long suma = 0;
            long skladnik = 0;
            while(skladnik <= limit)
            {
                suma += skladnik;
                skladnik++;
            }
            
            return suma;
        }
        
        public static int SumaParzystychFor(int limit) //DO SPRAWOZDANIA
        {
            int suma = 0;
            int licznik = limit;
            
            for (var i = 0; i < limit; i++)
            {
                if(licznik % 2 == 0)
                {
                    suma += licznik;
                }
                licznik --;
            }
            
            return suma;
        }

    }
